512  std::function<void(T&,const T&)> acc,
513  std::function<void(TensorT<T>&,const TensorT<T>&)> tensacc)
514 {
515  ade::Shape outshape = get_shape(out);
516  ade::NElemT nout = outshape.n_elems();
517  bool visited[nout];
518  std::memset(visited, false, nout);
519 
520  for (size_t i = 0, n = args.size(); i < n; ++i)
521  {
522  auto& arg = args[i];
523  if (is_identity(arg.mapper_))
524  {
525  if (i == 0)
526  {
527  out = *arg.data_;
528  std::memset(visited, true, nout);
529  }
530  else
531  {
532  tensacc(out, *arg.data_);
533  }
534  }
535  else // map coordinates
536  {
537  ade::CoordT coord;
538  ade::Shape inshape = get_shape(*arg.data_);
539 
540  T* ptr = out.data();
541  const T* argptr = arg.data_->data();
542  if (arg.push_)

544  for (ade::NElemT i = 0, n = inshape.n_elems(); i < n; ++i)
545  {
546  arg.mapper_->forward(coord.begin(),
547  ade::coordinate(inshape, i).begin());
548  ade::NElemT outidx = ade::index(outshape, coord);
549  if (visited[outidx])
